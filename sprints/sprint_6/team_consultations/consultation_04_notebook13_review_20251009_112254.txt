import Mathlib.Data.Finset.Basic

-- Define a permutation as a bijection on {1, ..., N}
def Permutation (N : ℕ) := Equiv (Fin N) (Fin N)

-- Define a descent in a permutation σ at position i if σ(i) > σ(i+1)
def isDescent {N : ℕ} (σ : Permutation N) (i : Fin (N-1)) : Prop :=
  σ.toFun i > σ.toFun (i + 1)

-- Define the set of descent positions for a permutation
def descentSet {N : ℕ} (σ : Permutation N) : Finset (Fin (N-1)) :=
  Finset.filter (isDescent σ) (Finset.univ)

-- Major index as sum of descent positions (1-based index)
def majorIndex {N : ℕ} (σ : Permutation N) : ℕ :=
  Finset.sum (descentSet σ) (λ i => i.val + 1)

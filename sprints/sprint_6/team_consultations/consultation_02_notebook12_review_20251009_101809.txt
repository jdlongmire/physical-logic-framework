import Mathlib.Data.Finset.Basic

-- Define the symmetric group S_N as permutations of {1, ..., N}
def SymmetricGroup (N : ℕ) := Equiv.Perm (Fin N)

-- Define adjacent transpositions as Coxeter generators
def AdjacentTransposition (N : ℕ) (i : Fin (N-1)) : SymmetricGroup N :=
  Equiv.swap (Fin.castSucc i) (Fin.succ i)

-- Define the Cayley graph for S_N with adjacent transpositions as generators
def CayleyGraph (N : ℕ) : SimpleGraph (SymmetricGroup N) :=
  SimpleGraph.fromRel (λ g h => ∃ i : Fin (N-1), h = g * (AdjacentTransposition N i))
